CREATE TABLE tipo (
	idtipo INT NOT NULL PRIMARY KEY AUTO_INCREMENT,
	nome VARCHAR(45) NOT NULL 
);

CREATE TABLE pokemon(
	idpokemon INT NOT NULL PRIMARY KEY AUTO_INCREMENT,
	nome VARCHAR(45) NOT NULL,
	codtipo INT,
	FOREIGN KEY (codtipo) REFERENCES tipo(idtipo)
);
